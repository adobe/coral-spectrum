/**

 */
function cssTimeToMilliseconds(time) {
  const num = parseFloat(time, 10);
  let unit = time.match(/m?s/);
 
  if (unit) {
    unit = unit[0];
  }
 
  if (unit === 's') {
    return num * 1000;
  } else if (unit === 'ms') {
    return num;
  }

  // unrecognized unit, so we return 0
  return 0;
}
